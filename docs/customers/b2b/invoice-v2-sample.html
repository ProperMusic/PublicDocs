Invoice v2.0 Sample

date format: YYYY-MM-DD
time format: HH:MM:SS
timestamp format: YYYY-MM-DD HH:MM:SS.ssssss
weight unit: grams
dimension unit: millimetre
invoicev2sample.xml

<?xml version="1.0" encoding="utf-8"?>
<transmission>
    <header>
        <transmissionControlNum>integer</transmissionControlNum>     # identifies this file
        <senderId>string</senderId>
        <senderGLN>string</senderGLN>
        <receiverId>string</receiverId>
        <receiverGLN>string</receiverGLN>
        <transmissionType>string</transmissionType>                 # value = INVOICES
        <transmissionCreatedDate>date</transmissionCreatedDate>
        <transmissionCreatedTime>time</transmissionCreatedTime>
        <countOfMessages>integer</countOfMessages>
        <customHeaders>
            <customerEANLocation/>
            <customerIdentityFromSupplier/>
            <customerName/>
            <supplierEANLocation/>
            <supplierIdentityFromCustomer/>
            <supplierName/>
        </customHeaders>
        <senderAddress>
            <name>PROPER MUSIC DISTRIBUTION LTD</name>
            <addressLine1>1-5 APPLEGARTH DRIVE</addressLine1>
            <addressLine2>QUESTOR</addressLine2>
            <addressLine3>DARTFORD</addressLine3>
            <addressCity>KENT</addressCity>
            <addressCountry>GB</addressCountry>
            <addressPostCode>DA1 1JD</addressPostCode>
            <vatRegistrationNum>GB627145936</vatRegistrationNum>
        </senderAddress>
    </header>
    <messages>
        <message>  # loop for each invoice included in the transmission
            <messageType>string</messageType>                        # value=INVOICE
            <messageControlNum>integer</messageControlNum>           # identifies this message
            <messageSequence>integer</messageSequence>               # ascending for each message in the transmission
            <invoiceNumber>string</invoiceNumber>
            <accountCode>string</accountCode>
            <shopNumber>integer</shopNumber>
            <billTo>
                <addressName>string</addressName>
                <addressLine1>string</addressLine1>
                <addressLine2>string</addressLine2>
                <addressLine3>string</addressLine3>
                <addressLine4>string</addressLine4>
                <city>string</city>
                <addressPostCode>string</addressPostCode>
                <countryCode>string</countryCode>
                <taxCode>string</taxCode>
                <taxName>string</taxName>
                <taxRate>decimal</taxRate>
                <taxRegistrationNum>string</taxRegistrationNum>
            </billTo>
            <deliverTo>
                <addressName>string</addressName>
                <addressLine1>string</addressLine1>
                <addressLine2>string</addressLine2>
                <addressLine3>string</addressLine3>
                <addressLine4>string</addressLine4>
                <city>string</city>
                <addressPostCode>string</addressPostCode>
                <countryCode>string</countryCode>
                <taxCode>string</taxCode>
                <taxName>string</taxName>
                <taxRate>decimal</taxRate>
                <taxRegistrationNum>string</taxRegistrationNum>
            </deliverTo>
            <invoiceDate>date</invoiceDate>
            <invoiceCreatedDate>date</invoiceCreatedDate>
            <invoiceCreatedTime>time</invoiceCreatedTime>
            <invoicedNetValue>decimal</invoicedNetValue>
            <invoicedTaxValue>decimal</invoicedTaxValue>
            <invoicedGrossValue>decimal</invoicedGrossValue>
            <currencyCode>string</currencyCode>
            <currencyName>string</currencyName>
            <currencyISOCode>string</currencyISOCode>
            <exchangeRate>decimal</exchangeRate>
            <countOfLines>integer</countOfLines>                         # lines in this invoice
            <sumOfUnits>integer</sumOfUnits>                             # total units in this invoice
            <orders>
                <order>         # loop for each order within this invoice (usually 1)
                    <sequence>integer</sequence>
                    <orderId>string</orderId>
                    <orderDate>date</orderDate>
                    <orderDocumentNo>string</orderDocumentNo>
                    <basilOrderNumber>string</basilOrderNumber>
                    <customerReferenceNo>string</customerReferenceNo>
                    <countOfLines>integer</countOfLines>         # lines in this order
                    <sumOfUnits>integer</sumOfUnits>             # total units in this order
                    <fulfilment>                                 # only for fulfilment clients
                        <clientId>integer</clientId>
                        <clientOrderId>string</clientOrderId>
                        <clientCustomerId>string</clientCustomerId>
                        <properOrderId>integer</properOrderId>
                        <properCustomerId>integer</properCustomerId>
                    </fulfilment>
                    <lines>         # only included if "Nested Lines" option set to true
                        <line>      # loop for each line in this order
                            <invoiceSequence>integer</invoiceSequence>   # ascending for each line within this invoice
                            <orderSequence>integer</orderSequence>       # ascending for each line within this order
                            <itemCode>string</itemCode>
                            <lineTotalExTax>decimal</lineTotalExTax>
                            <lineTaxValue>decimal</lineTaxValue>
                            <lineTotalIncTax>decimal</lineTotalIncTax>
                            <quantity>integer</quantity>
                            <itemValueExTax>decimal</itemValueExTax>
                            <itemTaxValue>decimal</itemTaxValue>
                            <itemValueIncTax>decimal</itemValueIncTax>
                            <taxCode>string</taxCode>
                            <taxRate>decimal</taxRate>
                        </line>
                    </lines>
                </order>
            </orders>
            <lines> # only included if "Nested Lines" option set to false
                <line> # loop for each line on this invoice
                    <invoiceSequence>integer</invoiceSequence> # ascending for each line within this invoice
                    <itemCode>string</itemCode>
                    <lineTotalExTax>decimal</lineTotalExTax>
                    <lineTaxValue>decimal</lineTaxValue>
                    <lineTotalIncTax>decimal</lineTotalIncTax>
                    <quantity>integer</quantity>
                    <itemValueExTax>decimal</itemValueExTax>
                    <itemTaxValue>decimal</itemTaxValue>
                    <itemValueIncTax>decimal</itemValueIncTax>
                    <taxCode>string</taxCode>
                    <taxRate>decimal</taxRate>
                    <orderId>string</orderId>
                    <orderDate>date</orderDate>
                    <orderDocumentNo>string</orderDocumentNo>
                    <basilOrderNumber>string</basilOrderNumber>
                    <customerReferenceNo>string</customerReferenceNo>
                    <fulfilment> # only included for fulfilment clients
                        <clientId>integer</clientId>
                        <clientOrderId>string</clientOrderId>
                        <clientCustomerId>string</clientCustomerId>
                        <properOrderId>integer</properOrderId>
                        <properCustomerId>integer</properCustomerId>
                    </fulfilment>
                </line>
            </lines>     
            <taxRates>
                <taxRate> # loop for each tax rate on this invoice
                    <sequence>integer</sequence> #ascending for each tax rate
                    <code>string</code>
                    <name>string</name>
                    <rate>decimal</rate>
                    <goodsValue>decimal<goodsValue>
                    <taxValue>decimal</taxValue>
                    <grossValue>decimal</grossValue>
                    <countOfLines>integer</countOfLines>
                    <sumOfUnits>integer</sumOfUnits>
                </taxRate>
            </taxRates>
            <countOfTaxRates>integer</countOfTaxRates>
            <invoiceTaxTotal>
                <amountExTax>decimal</amountExTax>
                <taxAmount>decimal</taxAmount>
                <amountIncTax>decimal</amountIncTax>
                <countOfLines>integer</countOfLines>
                <sumOfUnits>integer</sumOfUnits>
            </invoiceTaxTotal>
        </message>
    </messages>
    <footer>
        <taxRates> # all tax rates that appear in this transmission
            <taxRate> # loop for each tax rate in this transmission
                <sequence>integer</sequence>
                <code>string</code>
                <name>string</name>
                <rate>decimal</rate>
                <amountExTax>decimal</amountExTax>
                <taxAmount>decimal</taxAmount>
                <amountIncTax>decimal</amountIncTax>
                <countOfLines>integer</countOfLines>
                <sumOfUnits>integer</sumOfUnits>
            </taxRate>
        </taxRates>
        <taxTotal>
            <amountExTax>decimal</amountExTax>
            <taxAmount>decimal</taxAmount>
            <amountIncTax>decimal</amountIncTax>
            <countOfLines>integer</countOfLines>
            <sumOfUnits>integer</sumOfUnits>
        </taxTotal>
    </footer>
</transmission>
